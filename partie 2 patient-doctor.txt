// ========== Gestion des fichiers sélectionnés ==========
function addFiles(files) {
    const maxSize = 10 * 1024 * 1024; // 10 Mo
    const validFiles = [];

    for (let i = 0; i < files.length; i++) {
        const file = files[i];
        if (file.size > maxSize) {
            showToast(`${file.name} exceeds 10MB limit`, 'error');
        } else {
            validFiles.push(file);
        }
    }

    selectedFiles = selectedFiles.concat(validFiles);
    updateFileList();

    if (selectedFiles.length > 0) {
        document.getElementById('fileList').style.display = 'block';
        document.getElementById('recordInfoSection').style.display = 'block';
        document.getElementById('uploadBtn').disabled = false;
    }
}

function updateFileList() {
    const container = document.getElementById('selectedFiles');
    if (!container) return;

    let html = '';
    for (let i = 0; i < selectedFiles.length; i++) {
        const file = selectedFiles[i];
        html += `
            <div class="file-item">
                <div class="file-info">
                    <i class="fas ${getFileIcon(file.type)}"></i>
                    <div class="file-details">
                        <h4>${escapeHtml(file.name)}</h4>
                        <p>${formatFileSize(file.size)}</p>
                    </div>
                </div>
                <i class="fas fa-times remove-file" onclick="removeFile(${i})"></i>
            </div>
        `;
    }
    container.innerHTML = html;
}

function getFileIcon(mimeType) {
    if (!mimeType) return 'fa-file';
    if (mimeType.includes('pdf')) return 'fa-file-pdf';
    if (mimeType.includes('image')) return 'fa-file-image';
    if (mimeType.includes('text')) return 'fa-file-alt';
    return 'fa-file';
}

function formatFileSize(bytes) {
    if (bytes === 0) return '0 Bytes';
    const k = 1024;
    const sizes = ['Bytes', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
}

async function uploadFiles() {
    const recordTypeEl = document.getElementById('recordType');
    const recordDateEl = document.getElementById('recordDate');

    if (!recordTypeEl || !recordTypeEl.value) {
        showToast('Please select record type', 'error');
        return;
    }
    if (!recordDateEl || !recordDateEl.value) {
        showToast('Please select record date', 'error');
        return;
    }
    if (!currentUser) {
        showToast('User not authenticated', 'error');
        return;
    }

    // Afficher la progression
    document.getElementById('uploadProgress').style.display = 'block';
    document.getElementById('uploadBtn').disabled = true;
    document.getElementById('cancelBtn').style.display = 'inline-block';

    const totalFiles = selectedFiles.length;
    const uploadedFiles = [];

    try {
        for (let i = 0; i < selectedFiles.length; i++) {
            const file = selectedFiles[i];
            const progress = (i / totalFiles) * 100;
            const progressFill = document.getElementById('progressFill');
            const progressStatus = document.getElementById('progressStatus');

            if (progressFill) progressFill.style.width = progress + '%';
            if (progressStatus) progressStatus.textContent = `Uploading ${file.name} (${i + 1}/${totalFiles})`;

            const metadata = {
                recordType: recordTypeEl.value,
                recordDate: recordDateEl.value,
                description: document.getElementById('description')?.value || '',
                doctor: document.getElementById('doctor')?.value || '',
                tags: tags,
                uploadedBy: currentUser.did,
                uploadedAt: new Date().toISOString()
            };

            const result = await ipfsManager.uploadFile(file, metadata);
            if (result && result.success) {
                uploadedFiles.push({
                    cid: result.cid,
                    name: file.name,
                    size: file.size,
                    metadata: metadata
                });
            } else {
                throw new Error(result?.error || 'Upload failed');
            }
        }

        // Progression à 100%
        document.getElementById('progressFill').style.width = '100%';
        document.getElementById('progressStatus').textContent = 'Upload complete!';

        // Afficher le modal de succès pour le premier fichier
        if (uploadedFiles.length > 0) {
            showSuccessModal(uploadedFiles[0]);
        }

        // Réinitialiser après un délai
        setTimeout(resetUpload, 2000);

    } catch (error) {
        console.error('Upload error:', error);
        showToast('Upload failed: ' + error.message, 'error');
        document.getElementById('uploadProgress').style.display = 'none';
        document.getElementById('uploadBtn').disabled = false;
        document.getElementById('cancelBtn').style.display = 'none';
    }
}

async function confirmShare() {
    const doctorDidEl = document.getElementById('doctorDid');
    const accessDurationEl = document.getElementById('accessDuration');
    const shareCidEl = document.getElementById('shareRecordCid');
    const shareNameEl = document.getElementById('shareRecordName');

    if (!doctorDidEl || !accessDurationEl || !shareCidEl || !shareNameEl) {
        showError('Form elements not found');
        return;
    }

    const doctorDid = doctorDidEl.value.trim();
    const duration = parseInt(accessDurationEl.value) || 7;
    const cid = shareCidEl.value;
    const filename = shareNameEl.value;

    if (!doctorDid) return showError('Please enter doctor\'s DID');
    if (!cid) return showError('No record selected');

    showLoading();

    try {
        // Vérifier le médecin
        const isActive = await contractManager.verifyDoctor(doctorDid);
        if (!isActive) throw new Error('Doctor not found or not active');

        const doctorPubKey = await contractManager.getDoctorPublicKey(doctorDid);
        if (!doctorPubKey) throw new Error('Could not retrieve doctor\'s public key');

        const expiryTime = proxyManager.generateTimeRule(duration);
        const proxyResult = await proxyManager.requestReencryption(
            { cid, filename },
            doctorPubKey,
            expiryTime
        );

        if (!proxyResult || !proxyResult.success) {
            throw new Error(proxyResult?.error || 'Proxy encryption failed');
        }

        if (!currentUser || !currentUser.did) throw new Error('User not authenticated');

        // Enregistrer l'accès
        await contractManager.grantAccess(
            currentUser.did,
            doctorDid,
            cid,
            proxyResult.encryptedCid,
            expiryTime
        );

        // Envoyer une notification (optionnel, on ignore si la fonction n'existe pas)
        try {
            if (window.electronAPI.sendNotification) {
                await window.electronAPI.sendNotification({
                    toDid: doctorDid,
                    message: `Patient ${currentUser.name} has shared a document with you.`
                });
            }
        } catch (notifyError) {
            console.warn('Notification failed:', notifyError);
        }

        showSuccess('Record shared successfully');
        closeShareModal();

    } catch (error) {
        console.error('Share error:', error);
        showError(error.message || 'Failed to share record');
    } finally {
        hideLoading();
    }
}

async function loadStats() {
    const accesses = await window.electronAPI.storeGet('doctorAccesses:' + currentUser.did) || [];
    const now = Date.now() / 1000;
    let totalPatients = new Set();
    let activeAccesses = 0, expiringSoon = 0;
    for (let a of accesses) {
        if (a.patientDid) totalPatients.add(a.patientDid);
        if (a.expiryTime && a.expiryTime > now) {
            activeAccesses++;
            if (a.expiryTime - now < 7 * 24 * 60 * 60) expiringSoon++;
        }
    }
    return {
        totalPatients: totalPatients.size,
        availableRecords: accesses.length,
        activeAccesses,
        expiringSoon
    };

    const active = accesses.filter(a => a && a.expiryTime && a.expiryTime > now);

window.accessDocument = async function(transformedCtId) {
    if (!transformedCtId) {
        showError('No record selected');
        return;
    }
    showLoading('Decrypting...');
    try {
        // Call the proxy to decrypt and get the original CID
        const result = await proxyManager.decrypt(transformedCtId, currentUser.did);
        if (!result.cid) throw new Error(result.error || 'Decryption failed');
        const originalCid = result.cid;

        // Retrieve the file from IPFS
        const fileResult = await ipfsManager.getFile(originalCid);
        if (!fileResult.success) throw new Error('File not found on IPFS');

        const fileData = fileResult.data;
        // Convert base64 to blob and download or preview
        const blob = new Blob([ipfsManager.base64ToArrayBuffer(fileData.data)], { type: fileData.fileType || 'application/octet-stream' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = fileData.filename || 'document';
        a.click();
        URL.revokeObjectURL(url);

        showSuccess('Document retrieved');
    } catch (err) {
        console.error(err);
        showError(err.message);
    } finally {
        hideLoading();
    }
};

            PK['A'][attr] = PKa
            MK['A'][attr] = ska
        return (MK, PK, s, H)

    def registerUser(self, PK, H):
        sku = self.group.random()
        PKu = PK['P0'] ** sku
        mku = H['user'](PKu)
        return (sku, {'PKu': PKu, 'mku': mku})

    def hashDate(self, H, time, s):
        h = s
        key = 'y'
        if "year" in time:
            h = H['sy'](time['year']) ** h
        else:
            return None, None
        if "month" in time:
            h = H['sym'](time['month']) ** h
            key = 'ym'
            if "day" in time:
                h = H['symd'](time['day']) ** h
                key = 'ymd'
        elif "day" in time:
            return None, None
        return h, key

    def timeSuffices(self, timeRange, needle):
        if timeRange['year'] != needle['year']:
            return False
        if 'month' not in timeRange:
            return True
        if 'month' not in needle:
            return False
        if timeRange['month'] != needle['month']:
            return False
        if 'day' not in timeRange:
            return True
        if 'day' not in needle:
            return False
        return timeRange['day'] == needle['day']

    def keygen(self, MK, PK, H, s, user, pubuser, attribute, time):
        h, _ = self.hashDate(H, time, s)
        if h is None:
            return
        if 'SKu' not in user:
            user['SKu'] = PK['P0'] ** (MK['mk1'] * pubuser['mku'])
        PKat = PK['A'][attribute] * (PK['P0'] ** h)
        SKua = MK['SK1'] * (PKat ** (MK['mk1'] * pubuser['mku']))
        if 'A' not in user:
            user['A'] = {}
        if attribute not in user['A']:
            user['A'][attribute] = []
        user['A'][attribute].append((time, SKua))

    def encrypt(self, PK, policy, F):
        r = self.group.random()
        nA = lcm([len(term) for term in policy])
        U0 = PK['P0'] ** r
        U = []
        for term in policy:
            Ui = PK['P0'] ** 0
            for attr in term:
                Ui *= PK['A'][attr]
            U.append(Ui ** r)
        V = F * pair(PK['Q0'], PK['P1'] ** (r * nA))
        return {'A': policy, 'U0': U0, 'U': U, 'V': V, 'nA': nA}

    def reencrypt(self, PK, H, s, CT, currentTime):
        if 'year' not in currentTime or 'month' not in currentTime or 'day' not in currentTime:
            return None
        day = currentTime
        month = dict(day)
        del month['day']
        year = dict(month)
        del year['month']
        day_h, _ = self.hashDate(H, day, s)
        month_h, _ = self.hashDate(H, month, s)
        year_h, _ = self.hashDate(H, year, s)
        rs = self.group.random()
        U0t = CT['U0'] * (PK['P0'] ** rs)
        Ut = {'year': [], 'month': [], 'day': []}
        for term, Ui in zip(CT['A'], CT['U']):
            Uit_year = Ui
            Uit_month = Ui
            Uit_day = Ui
            for attr in term:
                Uit_year *= (PK['A'][attr] ** rs) * (U0t ** year_h)
                Uit_month *= (PK['A'][attr] ** rs) * (U0t ** month_h)
                Uit_day *= (PK['A'][attr] ** rs) * (U0t ** day_h)
            Ut['year'].append(Uit_year)
            Ut['month'].append(Uit_month)
            Ut['day'].append(Uit_day)
        Vt = CT['V'] * pair(PK['Q0'], PK['P1'] ** (rs * CT['nA']))
        return {'A': CT['A'], 'U0t': U0t, 'Ut': Ut, 'Vt': Vt, 'nA': CT['nA'], 't': currentTime}

    def decrypt(self, CT, user, term=None):
        if term is None:
            term = self.policyTerm(user, CT['A'])
            if term is False:
                return None
        sumSK = 1
        for attr in CT['A'][term]:
            found = False
            for timeRange, SKua in user['A'][attr]:
                if self.timeSuffices(timeRange, CT['t']):
                    sumSK *= SKua
                    found = True
                    break
            if not found:
                return None
        n = CT['nA'] // len(CT['A'][term])
        return CT['Vt'] / (pair(CT['U0t'], sumSK ** n) / pair(user['SKu'], CT['Ut']['year'][term] ** n))

    def policyTerm(self, user, policy):
        userAttrs = list(user['A'].keys())
        for i, term in enumerate(policy):
            ok = True
            for attr in term:
                if attr not in userAttrs:
                    ok = False
                    break
            if ok:
                return i
        return False

# Storage for doctor keys (by DID)
doctor_keys = {}

# Helper to generate a deterministic GT from a string (for CID)
def string_to_GT(s):
    # Use SHA256 to get a hash, then map to GT via group.hash
    h = hashlib.sha256(s.encode()).digest()
    # group.hash expects bytes and returns a GT element
    return group.hash(h, GT)

@app.route('/doctor/register', methods=['POST'])
def doctor_register():
    """Register a doctor: generate attribute keys for a given set of attributes and time validity."""
    data = request.json
    doctor_did = data['doctor_did']
    attributes = data.get('attributes', ["doctor"])   # default attribute "doctor"
    time_range = data.get('time_range', {
        'year': '2025',
        'month': '3',
        'day': '29'   # for demo, use current date
    })
    # Register the doctor as a user in the TBPRE system
    sku, pubuser = tbpre.registerUser(PK, H)
    user = {
        'id': doctor_did,
        'sku': sku,
        'pubuser': pubuser,
        'SKu': PK['P0'] ** (MK['mk1'] * pubuser['mku']),
        'A': {}
    }
    for attr in attributes:
        tbpre.keygen(MK, PK, H, s, user, pubuser, attr, time_range)
    doctor_keys[doctor_did] = user
    return jsonify({'success': True, 'message': f'Doctor {doctor_did} registered with attributes {attributes}'})

@app.route('/encrypt', methods=['POST'])
def encrypt():
    data = request.json
    cid = data['cid']
    policy = data['policy']   # e.g. [["cardiology"]]
    F = string_to_GT(cid)     # map CID to a GT element
    CT = tbpre.encrypt(PK, policy, F)
    # Serialize ciphertext
    def serialize(g):
        return group.serialize(g).hex()
    serialized = {
        'A': CT['A'],
        'U0': serialize(CT['U0']),
        'U': [serialize(u) for u in CT['U']],
        'V': serialize(CT['V']),
        'nA': CT['nA']
    }
    return jsonify({'success': True, 'ciphertext': serialized, 'original_cid': cid})

@app.route('/reencrypt', methods=['POST'])
def reencrypt():
    data = request.json
    ciphertext = data['ciphertext']
    current_date = data['current_date']
    # Deserialize
    def deserialize(hex_str, type_class):
        return group.deserialize(bytes.fromhex(hex_str))
    CT = {
        'A': ciphertext['A'],
        'U0': deserialize(ciphertext['U0'], G1),
        'U': [deserialize(u, G1) for u in ciphertext['U']],
        'V': deserialize(ciphertext['V'], GT),
        'nA': ciphertext['nA']
    }
    CTt = tbpre.reencrypt(PK, H, s, CT, current_date)
    if CTt is None:
        return jsonify({'success': False, 'error': 'Invalid date'})
    # Serialize
    def serialize(g):
        return group.serialize(g).hex()
    serialized = {
        'A': CTt['A'],
        'U0t': serialize(CTt['U0t']),
        'Ut': {
            'year': [serialize(u) for u in CTt['Ut']['year']],
            'month': [serialize(u) for u in CTt['Ut']['month']],
            'day': [serialize(u) for u in CTt['Ut']['day']]
        },
        'Vt': serialize(CTt['Vt']),
        'nA': CTt['nA'],
        't': CTt['t']
    }
    return jsonify({'success': True, 'reencrypted_ciphertext': serialized})

@app.route('/decrypt', methods=['POST'])
def decrypt():
    data = request.json
    reencrypted = data['reencrypted_ciphertext']
    doctor_did = data['doctor_did']
    if doctor_did not in doctor_keys:
        return jsonify({'success': False, 'error': 'Doctor not registered. Please register first.'})
    user = doctor_keys[doctor_did]
    # Deserialize
    def deserialize(hex_str, type_class):
        return group.deserialize(bytes.fromhex(hex_str))
    CTt = {
        'A': reencrypted['A'],
        'U0t': deserialize(reencrypted['U0t'], G1),
        'Ut': {
            'year': [deserialize(u, G1) for u in reencrypted['Ut']['year']],
            'month': [deserialize(u, G1) for u in reencrypted['Ut']['month']],
            'day': [deserialize(u, G1) for u in reencrypted['Ut']['day']]
        },
        'Vt': deserialize(reencrypted['Vt'], GT),
        'nA': reencrypted['nA'],
        't': reencrypted['t']
    }
    F = tbpre.decrypt(CTt, user)
    if F is None:
        return jsonify({'success': False, 'error': 'Decryption failed – time condition not met or attributes mismatch'})
    # We need to retrieve the original CID from F. Since we mapped CID to GT element, we cannot reverse it.
    # To solve this, we store a mapping (ciphertext_hash -> original_cid) in the server during encryption.
    # For simplicity, we require the client to send the original_cid in the request.
    original_cid = data.get('original_cid')
    if not original_cid:
        return jsonify({'success': False, 'error': 'original_cid missing'})
    return jsonify({'success': True, 'original_cid': original_cid})
